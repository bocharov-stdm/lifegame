simulation_project/
├── main.py                   # Главный файл, запускает симуляцию
├── settings.py               # Все параметры симуляции (например, k_speed, размеры поля и т.п.)

├── organisms/
│   ├── __init__.py
│   ├── base.py              # Базовый класс Organism
│   ├── herbivore.py         # Травоядное
│   ├── carnivore.py         # Хищник

├── environment/
│   ├── __init__.py
│   ├── food.py              # Класс еды
│   ├── map.py               # Карта или поле, границы, возможно препятствия

├── ui/
│   ├── __init__.py
│   ├── sliders.py           # GUI элементы для настройки (скорость, популяция и т.п.)
│   ├── debug_panel.py       # Панель информации: FPS, численность, параметры организмов

├── simulation/
│   ├── __init__.py
│   ├── engine.py            # Основная логика симуляции (обновление всех объектов, взаимодействие)
│   ├── time.py              # Время симуляции (тики, ускорение/пауза)

├── assets/                  # Ресурсы: изображения, шрифты, иконки

└── utils/
    ├── __init__.py
    ├── math_helpers.py      # Помощники для векторов, расстояний, нормализации
    └── colors.py            # Цветовые палитры, возможно визуализация характеристик
